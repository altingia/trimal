41 /**
42  \brief Abstract class that serves as template for Format Handlers.
43  Formats must inherit from this class and
44  will be handled by \link FormatManager \endlink
45 
46  Each class should know how to:
47  - Recognize if a file is in its format\n
48  - Load a file in its format and return an alignment object\n
49  - Save an alignment to a file in this format\n
50  - Recognize its format acronyms.
51  */
52 class BaseFormatHandler {
53 public:
54 
55  /** \brief Bool tag that indicates whether or not this state
56  * can load an alignment in the corresponding format.*/
57  bool canLoad = false;
58 
59  /** \brief Bool tag that indicates whether or not this state
60  * can save an alignment in the corresponding format.*/
61  bool canSave = false;
62 
63  /**
64  \brief String that contains a well known acronym to the format
65  */
66  std::string name;
67 
68  /**
69  \brief String that contains the main extension of the format
70  */
71  std::string extension;
72 
73  /**
74  \brief Function to check if a file is in the implemented format.
75  \param origin File Handler to the file.
76  \return A number that represents the confidence recognizing this format.
77  \attention Take in mind incompatibilities with
78  other formats when you implement this function:\n\n
79  <i>Fasta may recognize a file if it starts with '>', returning a 1.\n
80  Pir files also begin with '>', but they also recognize the ';' after the '>' and 2 letter symbol.\n
81  When the Pir recognizes its format, it returns a 2.\n
82  So, when the Pir format is recognized, it's preferred over the Fasta Format,
83  when both of them can load the alignment bur pir extracts more information.</i>\n
84  */
85  virtual int CheckAlignment(std::istream *origin) = 0;
86 
87  /**
88  \brief Function to load a file in the current format and return an alignment object.
89  \param filename Filename of the file to load.
90  \return <b> Alignment</b> loaded with the information of the file. \n
91  <b> nullptr</b> if there was any error.
92  */
93  virtual Alignment *LoadAlignment(const std::string &filename) = 0;
94 
95  /**
96  \brief Function to save a \link Alignment \endlink to a file.
97  \param alignment Alignment to save.
98  \param output File Handler where to save the formatted alignment;
99  \return <b> True </b>if file could be saved.\n
100  <b> False </b> otherwise.
101  */
102  virtual bool SaveAlignment(const Alignment &alignment, std::ostream *output) = 0;
103 
104  /**
105  \brief Function that recognizes acronyms of the format.
106  \param FormatName acronym to recognize
107  \return <b>True</b> if recognized\n
108  <b>False</b> otherwise.
109  */
110  virtual bool RecognizeOutputFormat(const std::string &FormatName) {
111  return (name == FormatName);
112  }
113 
114  /**
115  \brief Class Destructor
116  */
117  virtual ~BaseFormatHandler() = default;
118 
119 protected:
120  /**
121  \brief Pointer to the \link FormatManager \endlink
122  that contains and manages this state
123  */
124  FormatHandling::FormatManager *Machine;
125 };
