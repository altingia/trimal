36  int getTagFromCharArray(
37  const char *const array,
38  const char separator)
39  {
40  if (array[1] != separator)
41  return -1;
42  // Loop variables
43  size_t c, maxlen;
44  // Current tag value
45  int curval = 0;
46  // Iterate over the sequence, bit-adding the tag
47  for (c = 0, maxlen = strlen(array); c < maxlen; c++) {
48  switch (array[c]) {
49  case 'A':
50  curval |= 1 << 0;
51  break; // = 1
52  case 'C':
53  curval |= 1 << 1;
54  break; // = 2
55  case 'T':
56  curval |= 1 << 2;
57  break; // = 4
58  case 'G':
59  curval |= 1 << 3;
60  break; // = 8
61  default:
62  break;
63  }
64 
65  // Check if we haven't arrived to end of sequence
66  if (++c < maxlen)
67  {
68  // check the next element in array is the separator char
69  if (array[c] == separator)
70  continue;
71  // Otherwise, it's not a well-formed sequence to convert
72  else
73  return -1;
74  }
75  }
76  // If we have arrived to the end of the sequence
77  // it was well formed and parsed.
78  if (c == maxlen)
79  return curval;
80  // Otherwise, we have early stopped.
81  return -1;
82  }

84  char getCharFromTag(const int tag) {
85  switch (tag) {
86 
87  // One Base
88  // case 1://A = 1
89  // case 2://C = 2
90  // case 4://T = 4
91  // case 8://G = 8

94  case 3:

100  case 6:
101  return 'Y';// pYrimidine CT
102  case 10:
103  return 'S';// Strong CG
104  case 12:
105  return 'K';// Keto TG
106

108  case 14:
109  return 'B';// CTG not A

111  return 'D';// ATG not C
112  case 11:
113  return 'V';// ACG not T
114  case 7:
115  return 'H';// ACT not G
116 
117  // All Four Bases

142  inline bool checkContigsWithReference(
143  const AlignmentVector &sources,
144  const StringVector &contigs)
145  {
146  // Store the result value, as we want to report all errors
147  // and thus, not skipping on first error
148  bool checkIn = true;
149  // Iterate over the list of contigs
150  for (const std::string & contig : contigs) {
151  int i;
152  // Iterate all over the sources
153  for (i = 0; i < sources.size(); i++) {
154  // And early skip on first coincidence
155  if (!strcmp(&contig[0], &sources[i]->seqsName[0][0]))
156  break;
157  }
158  // If we haven't early skipped, then, contig doesn't have
159  // an alignment assigned, and thus, there is no
160  // sequence on the original alignment that correspond
161  // to the contig.
162  if (i == sources.size()) {
163  debug.report(ErrorCode::NoReferenceSequenceForContig, &contig[0]);
164  checkIn = false;
165  }
166  }
167  // Return the result value, which is false in case any contig
168  // doesn't have a reference sequence on the original alignment
169  return checkIn;

212  inline bool extractFeature(
213  const char *const line,
214  vcfFeature &feature)
215  {
216  if (line[0] == '#')
217  return false;
218 
219  std::string tmpLine{line};
220  char *tmp;
221 
222  size_t strlenVal;
223 
224  feature.reset();
225 
226  // Contig
227  tmp = std::strtok(&tmpLine[0], "\t");
228  strlenVal = strlen(tmp);
229  feature.contig = new char[strlenVal + 1];
230  std::memmove(feature.contig, tmp, strlenVal);

232

234  tmp = std::strtok(nullptr, "\t");
235  feature.position = atoi(tmp) - 1;

237  // ID

246 
247  // Alt
248  tmp = std::strtok(nullptr, "\t");
249  strlenVal = strlen(tmp);

251  std::memmove(feature.alt, tmp, strlenVal);
252  feature.alt[strlenVal] = '\0';
253

257  if (alt != -1)

261  }
262  }
263 
264  // Quality
265  tmp = std::strtok(nullptr, "\t");
266  feature.quality = (float) atof(tmp);
267 
268  // Filter
269  tmp = std::strtok(nullptr, "\t");
270  feature.filter = std::strcmp(tmp, "PASS") == 0;
271 
272  // Info
273  std::strtok(nullptr, "\t");
274 
275  // Format
276  tmp = std::strtok(nullptr, "\t");
277  char *format = new char[strlen(tmp) + 1];
278  std::memmove(format, tmp, strlen(tmp) + 1);
279 
280  // Donors
281  int counter = 0;
282  tmp = std::strtok(nullptr, "\t");
283  while (tmp != nullptr) {
284  feature.donorsInfo.emplace_back(tmp);
285  tmp = std::strtok(nullptr, "\t");
286  }
287 
288  // Format -> Read Depth Index
289  counter = 0;
290  tmp = std::strtok(format, ":");
291  feature.readDepthIndex = -1;
292  while (tmp != nullptr) {
293  if (strlen(tmp) > 1 && tmp[0] == 'D' && tmp[1] == 'P') {
294  feature.readDepthIndex = counter;
295  break;
296  }
297  tmp = std::strtok(nullptr, ":");
298  counter++;
299  }
300 
301  delete[] format;
302 
303  return true;
304  }

322  inline void applyVariantCallingFiles(
323  const AlignmentVector &sources,
324  const StringVector &filenames,
325  const StringVector &contigs,
326  const std::vector<std::vector<int>> &donorsPositions,
327  const float minQuality,
328  const float minCoverage,
329  const bool ignoreFilter,
330  const char *const replacementChar)
331  {
332  char *line = new char[4096];
333 
334  // Create a feature object to reuse
335  vcfFeature feature = vcfFeature();
336 
337  for (int vcfIndex = 0; vcfIndex < filenames.size(); vcfIndex++) {
338 
339  // Open the VCF file
340  std::ifstream infile(filenames[vcfIndex]);
341  if (!infile.good()) {
342  debug.report(ErrorCode::CantOpenFile, &filenames[vcfIndex][0]);
343  continue;
344  }
345 
346  // Iterate over all VCF
347  while (infile.getline(line, 4096, '\n')) {
348 
349  // Extract feature line by line
350  if (!extractFeature(line, feature))
351  // Continue if line is not feature
352  continue;
353 
354 
355  // Check if feature-referred contig is found
356 
357  // contigIndex is also sourcesIndex
358  int contigIndex;
359  {
360  for (contigIndex = 0; contigIndex < contigs.size(); contigIndex++)
361  {
362  if (contigs[contigIndex] == feature.contig)
363  break;
364  }
365  if (contigIndex == contigs.size()) {
366  std::cout << "Contig not Found" << std::endl;
367  continue;
368  }
369  }
370 
371  // Get the alignment that refers to the contig
372  Alignment * alig = sources[contigIndex];
373 
374  // Cannot apply the feature if the position to change
375  // is bigger than the alignment
376  if (alig->originalNumberOfResidues <= feature.position)
377  {
378  debug.report(ErrorCode::SNPoutOfBounds,
379  new std::string[3]
380  {
381  std::to_string(feature.position),
382  filenames[vcfIndex],
383  std::to_string(alig->originalNumberOfResidues)
384  }
385  );
386  continue;
387  }
388 
389  // Filter-out features with low quality or not snp
390  if (!filterFeature(feature, minQuality))
391  continue;
392 
393  // Iterate over all donors
394  for (int donorID = 0; donorID < feature.donorsInfo.size(); donorID++)
395  {
396  // Sanity check to prevent OutOfBounds
397  if (donorID >= donorsPositions[vcfIndex].size())
398  {
399  debug.report(ErrorCode::MoreDonorsOnLineThanPresented,
400  new std::string[1]{ filenames[vcfIndex]} );
401  continue;
402  }
403  // Get the sequence position.
404  const int seqPos = donorsPositions[vcfIndex][donorID];
405 
406  // Get the character & reference for further use
407  char & donorSequenceChar =
408  alig->sequences[seqPos][feature.position];
409  const char & referenceSequenceChar =
410  alig->sequences[0][feature.position];
411 
412  // In case we filter - out by VCF filter, apply the replacement char
413  if (!feature.filter && !ignoreFilter)
414  {
415  if (replacementChar) {
416  alig->sequences[seqPos][feature.position] = *replacementChar;
417  }
418  continue;
419  }
420 
421  // Get the string that informs the user whether
422  // a feature should be applied to a donor
423  const std::string & referenceString = feature.donorsInfo[donorID];
424 
425  // Only apply if the string is valid - different to 0
426  if (referenceString != "0") {
427  std::string donorInfo{referenceString};
428  // Extract the read depth
429  char *tmp = std::strtok(&donorInfo[0], ":");
430  for (int V = 0; V < feature.readDepthIndex; V++)
431  tmp = std::strtok(nullptr, ":");
432  int readDepth = std::stoi(tmp);
433 
434  // Filter out by read depth
435  if (readDepth < minCoverage) {
436  // If provided, change the character by the replacement-char
437  // as we don't know if the character is equal to the
438  // reference or the provided SNP
439  if (replacementChar)
440  donorSequenceChar = *replacementChar;
441  break;
442  }
443 
444  // Check if the reference character is the same as expected in the sequence to apply
445  if (donorSequenceChar == std::tolower(feature.ref[0]) ||
446  donorSequenceChar == std::toupper(feature.ref[0]))
447  {
448  debug.report(InfoCode::AddingSNP,
449  new std::string[5]
450  {
451  contigs[contigIndex],
452  alig->seqsName[seqPos],
453  std::to_string(feature.position),
454  std::string(1, feature.ref[0]),
455  std::string(1, feature.alt[0])
456  });
457  donorSequenceChar = feature.alt[0];
458  }
459  // Check if the alternative sequence is the same in the sequence to apply
460  else if (donorSequenceChar == feature.alt[0])
461  {
462  debug.report(WarningCode::SNPAlreadApplied,
463  new std::string[5]
464  {
465  contigs[contigIndex],
466  alig->seqsName[seqPos],
467  std::to_string(feature.position),
468  std::string(1, feature.ref[0]),
469  std::string(1, feature.alt[0])
470  });
471  }
472  // Check if reference character is the same as expected in the reference sequence
473  else if (referenceSequenceChar == std::tolower(feature.ref[0]) ||
474  referenceSequenceChar == std::toupper(feature.ref[0]))
475  {
476  debug.report(ErrorCode::OverwrittingSNP,
477  new std::string[7]
478  {
479  contigs[contigIndex],
480  alig->seqsName[seqPos],
481  std::to_string(feature.position),
482  std::string(1, referenceSequenceChar),
483  std::string(1, feature.alt[0]),
484  std::string(1, referenceSequenceChar),
485  std::string(1, donorSequenceChar),
486  });
487  donorSequenceChar = feature.alt[0];
488  }
489  else {
490  debug.report(ErrorCode::ReferenceNucleotideNotCorresponding,
491  new std::string[5]
492  {
493  contigs[contigIndex],
494  alig->seqsName[donorID],
495  std::to_string(feature.position),
496  std::string(1, feature.ref[0]),
497  std::string(1, referenceSequenceChar)
498  });
499  }
500  }
501 
502  }
503  }
504  }
505  delete[] line;
506  }

509  inline void extractContigsFromLine(
510  char *const line,
511  StringVector &contigs)
512  {
513  // Remove first two characters;
514  memmove(line, line + 2, strlen(line + 2) + 2);
515  // Print result
516  char *field_name = std::strtok(line, "=");
517 
518  // We only want the contig fields
519  if (!strcmp(field_name, "contig")) {
520  std::strtok(nullptr, "=");
521  char *field_info = std::strtok(nullptr, ",");
522  char *fname = new char[std::strlen(field_info) + 1];
523  memmove(fname, field_info, strlen(field_info));
524  fname[std::strlen(field_info)] = '\0';
525 
526  int U;
527  // Check if the contig has already been added.
528  for (U = 0; U < contigs.size(); U++)
529  if (contigs[U] == fname)
530  break;
531 
532  // If not, add it to the vector.
533  if (U == contigs.size())
534  contigs.emplace_back(fname);
535 
536  delete[] fname;
537  }
538  }

540  inline void extractDonorsFromLine(
541  char *const line,
542  std::vector<int> &currentDonorPosition,
543  StringVector &donors)
544  {
545  // We only want to parse the FORMAT line, which contains the donors.
546  std::strtok(strstr(line, "FORMAT"), "\t");
547 
548  char *token = std::strtok(nullptr, "\t");
549  while (token != nullptr) {
550  char *fname = new char[std::strlen(token) + 1];
551  memmove(fname, token, strlen(token));
552  fname[std::strlen(token)] = '\0';
553 
554  int U;
555 
556  // Check every other donor already added.
557  for (U = 0; U < donors.size(); U++) {
558  if (donors[U] == fname) {
559  break;
560  }
561  }
562 
563  // If not present, we add it
564  if (U == donors.size()) {
565  donors.emplace_back(fname);
566  }
567 
568  // If already present, warn the user.
569  else {
570  debug.report(WarningCode::DonorAlreadyAdded, &fname[0]);
571  }
572 
573  // We sum 1 as the position 0 is for reference
574  currentDonorPosition.emplace_back(U + 1);
575 
576  token = std::strtok(nullptr, "\t");
577  delete[] fname;
578  }
579  }

581  inline void obtainContigsAndDonors(
582  const StringVector &filenames,
583  StringVector &donors,
584  StringVector &contigs,
585  std::vector<std::vector<int>> &donorsPositions)
586  {
587  const int bufSize = 4096;
588  char *line = new char[bufSize];
589 
590  // Obtain contigs and donors from all VCF files.
591  for (int C = 0; C < filenames.size(); C++) {
592 
593  // Add a new vector<int>, which holds the sequence index
594  // for all donors present on this VCF
595  donorsPositions.emplace_back(std::vector<int>());
596 
597  // Open the VCF file
598  std::ifstream infile(filenames[C]); // Deleted with RAII
599 
600  if (!infile.good()) {
601  debug.report(ErrorCode::CantOpenFile, &filenames[C][0]);
602  continue;
603  }
604 
605  // Read file to get all the donors and contigs present on the VCF
606  while (infile.getline(line, bufSize, '\n')) {
607  // Extract contigs
608  if (strncmp("##contig", line, 8) == 0)
609  extractContigsFromLine(line, contigs);
610  // Extract donors
611  else if (strncmp("#CHROM", line, 6) == 0)
612  extractDonorsFromLine(line, donorsPositions[C], donors);
613  }
614  }
615  delete[] line;
616  }

619  void readVCF(
620  const AlignmentVector &sources,
621  const StringVector &filenames,
622  const float minQuality,
623  const float minCoverage,
624  const bool ignoreFilter,
625  const char *const replacementChar)
626  {
627  // All donors present in the vcf files.
628  StringVector donors = StringVector();
629  // Contigs present on the VCF files. Each entry must be present in the sources vector.
630  StringVector contigs = StringVector();
631  // Position on the reference files of each donor.
632  // Donors positions is a vector, of vectors, of ints.
633  // The first level refers to the VCF index in filenames
634  // The second level refers to the donor index in the VCF
635  // The value obtained is the sequence index of that donor in an alignment.
636  std::vector<std::vector<int>> donorsPositions = std::vector<std::vector<int>>();
637 
638  {
639  using namespace ngs::__internal;
640 
641  // Get a list of donors, contigs and a map, called donorsPositions
642  obtainContigsAndDonors(filenames, donors, contigs, donorsPositions);
643 
644  // Check if all contigs have a reference alignment.
645  if (!checkContigsWithReference(sources, contigs))
646  return;
647 
648  // Extend the reference files with new sequences from donors.
649  increaseSequencesInAlignment(sources, donors);
650 
651  applyVariantCallingFiles(
652  sources, filenames, contigs, donorsPositions,
653  minQuality, minCoverage, ignoreFilter, replacementChar);
654  }
655  }
