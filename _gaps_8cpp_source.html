93  bool Gaps::applyWindow(int halfW) {
94  // Create a timerLevel that will report times upon its destruction
95  // which means the end of the current scope.
96  StartTiming("bool Gaps::applyWindow(int halfW) ");
97 
98  if (halfW > alig->originalNumberOfResidues / 4) {
99  debug.report(ErrorCode::GapWindowTooBig);
100  return false;
101  }
102 
103  // Save the requested half window. This is useful when making a copy of the
104  // alignment, as the window values are not valid anymore but don't want to
105  // calculate them if not needed anymore
106  halfWindow = halfW;
107 
108  // If the half window requested is 0 or a negative number
109  // we simply delete the window values.
110  if (halfW < 1) {
111 
112  delete[] gapsWindow;
113  gapsWindow = nullptr;
114  return true;
115  }
116 
117  int i, j, window;
118 
119  if (gapsWindow == nullptr)
120  gapsWindow = new int[alig->originalNumberOfResidues];
121 
122  // Initialize to 0 the vector that will store the number of gaps of each column
123  // and the vector that will store window processing results
124  utils::initlVect(numColumnsWithGaps, alig->originalNumberOfSequences + 1, 0);
125 
126  // Initialize maximum gaps number per column value and store the mediumWinSize value in the object.
127  maxGaps = 0;
128  window = (2 * halfWindow + 1);
129 
130  // We calculate some statistics for every column in the alignment,and the maximum gaps' number value
131  for (i = 0; i < alig->originalNumberOfResidues; i++) {
132  gapsWindow[i] = 0;
133  // Sum the total number of gaps for the considered window
134  for (j = i - halfWindow, gapsWindow[i] = 0; j <= i + halfWindow; j++) {
135  if (j < 0)
136  gapsWindow[i] += gapsInColumn[-j];
137  else if (j >= alig->originalNumberOfResidues)
138  gapsWindow[i] += gapsInColumn[((2 * alig->originalNumberOfResidues - j) - 2)];
139  else
140  gapsWindow[i] += gapsInColumn[j];
141  }
142 
143  // Calculate, and round to the nearest integer, the number of gaps for the i column
144  gapsWindow[i] = utils::roundInt(((double) gapsWindow[i] / window));
145  // Increase in 1 the number of colums with the same number of gaps than column i
146  numColumnsWithGaps[gapsWindow[i]]++;
147 
148  // Update the max number of gaps in the alignment, if neccesary
149  if (gapsWindow[i] > maxGaps)
150  maxGaps = gapsWindow[i];
151  }
152  return true;
153  }

181  double Gaps::calcCutPoint(float minInputAlignment, float gapThreshold) {
182  // Create a timerLevel that will report times upon its destruction
183  // which means the end of the current scope.
184  StartTiming("double Gaps::calcCutPoint(float minInputAlignment, float gapThreshold) ");
185 
186  // Method to select the cutting point based on gaps values from the input
187  // alignment. The cutting point is selected as the maximum gaps number allowed
188  // in the output alignment given a minimum percentage of the input alignment
189  // to be kept and a maximum gaps number. In case of both values set different
190  // cutting points, the minimum percentage of the input alignment prevails.
191 
192  double cuttingPoint_MinimumConserv, cuttingPoint_gapThreshold;
193  int i, acum;
194 
195  // Calculate the gap number represented by the gaps threshold. This gap number
196  // represents the maximum gap number in any column in the output alignment
197  cuttingPoint_gapThreshold = (double) alig->numberOfSequences * gapThreshold;
198 
199  // Compute the minimum columns number to be kept from the input alignment
200  cuttingPoint_MinimumConserv = utils::roundInt(((double) (alig->originalNumberOfResidues * minInputAlignment) / 100.0));
201  if (cuttingPoint_MinimumConserv > alig->originalNumberOfResidues)
202  cuttingPoint_MinimumConserv = alig->originalNumberOfResidues;
203 
204  // We look at the number of gaps which allows us to keep the minimum columns
205  // number from the input alignment
206  for (i = 0, acum = 0; i < alig->originalNumberOfSequences; i++) {
207  acum += numColumnsWithGaps[i];
208  if (acum >= cuttingPoint_MinimumConserv)
209  break;
210  }
211 
212  // If there is not an exact number for the gaps cutting point, compute such
213  // value as the inmediate superior gap number minus the proportion of columns
214  // necessary to respect the minimum percentage from the input alignment to be
215  // kept
216  if (numColumnsWithGaps[i])
217  cuttingPoint_MinimumConserv =
218  (double) (i - ((float) (acum - cuttingPoint_MinimumConserv) / numColumnsWithGaps[i]));
219  else
220  cuttingPoint_MinimumConserv = 0;
221 
222  // Return the maximum gap number of the two computed cutting points.
223  return (utils::max(cuttingPoint_MinimumConserv, cuttingPoint_gapThreshold));
224  }

226  int Gaps::calcCutPointMixSlope() {
227  // Create a timerLevel that will report times upon its destruction
228  // which means the end of the current scope.
229  StartTiming("int Gaps::calcCutPointMixSlope(void) ");
230 
231  float delta = 0, maxSlope = -1, *firstSlopeVector, *secondSlopeVector;
232  int prev, pprev, maxIter, row = 1, act = 0, max = 0;
233 
234  // We build two slope vectors, one vector for the first slope and another one for the second.
235  firstSlopeVector = new float[maxGaps + 1];
236  secondSlopeVector = new float[maxGaps + 1];
237 
238  // We initialize them with -1.0 value and fix the maximum iteractions' number as maximun gaps' number plus 1
239  utils::initlVect(firstSlopeVector, maxGaps, -1.0F);
240  utils::initlVect(secondSlopeVector, maxGaps, -1.0F);
241  maxIter = maxGaps + 1;
242 
243  // Until to achieve the maximum iteractions' number.
244  while (act < maxIter) {
245 
246  // We look for a first point to second slope.
247  while ((numColumnsWithGaps[act]) == 0) act++;
248  pprev = act;
249  if ((act + 1) >= maxIter) break;
250 
251  // We look for a first point to first slope.
252  do {
253  act++;
254  } while ((numColumnsWithGaps[act]) == 0);
255  prev = act;
256  if ((act + 1) >= maxIter) break;
257 
258  // We look for a second point to first and second slope.
259  do {
260  act++;
261  } while ((numColumnsWithGaps[act]) == 0);
262  if (act >= maxIter) break;
263 
264  // Calculate the first slope between the earlier previus and previus points.
265  firstSlopeVector[prev] = ((float) (prev - pprev) / alig->originalNumberOfSequences);
266  firstSlopeVector[prev] /= ((float) numColumnsWithGaps[prev] / alig->originalNumberOfResidues);
267 
268  // Calculate the first slope between the previus and current points.
269  firstSlopeVector[act] = ((float) (act - prev) / alig->originalNumberOfSequences);
270  firstSlopeVector[act] /= ((float) numColumnsWithGaps[act] / alig->originalNumberOfResidues);
271 
272  // Calculate the second slope between the earlier previus and current points.
273  secondSlopeVector[act] = ((float) (act - pprev) / alig->originalNumberOfSequences);
274  secondSlopeVector[act] /= ((float) (numColumnsWithGaps[act] + numColumnsWithGaps[prev]) / alig->originalNumberOfResidues);
275 
276  // If the ratio between ...
277  if ((secondSlopeVector[pprev] != -1.0) || (firstSlopeVector[pprev] != -1.0)) {
278 
279  // first slope previus and first slope earlier previus points.
280  if (firstSlopeVector[pprev] != -1.0) {
281  delta = firstSlopeVector[prev] / firstSlopeVector[pprev];
282  row = pprev;
283  }
284 
285  // first slope current and first slope previus points.
286  if (delta < (firstSlopeVector[act] / firstSlopeVector[prev])) {
287  delta = firstSlopeVector[act] / firstSlopeVector[prev];
288  row = prev;
289  }
290 
291  // second slope current and second slope earlier previus points.
292  if (secondSlopeVector[pprev] != -1.0) {
293  if (delta < (secondSlopeVector[act] / secondSlopeVector[pprev])) {
294  delta = secondSlopeVector[act] / secondSlopeVector[pprev];
295  row = pprev;
296  }
297  }
298 
299  // ... is better that current maxSlope then we modify the maxSlope with the best ratio.
300  if (delta > maxSlope) {
301  maxSlope = delta;
302  max = row;
303  }
304  }
305  act = prev;
306  }
307 
308  // We delete the local memory.
309  delete[] firstSlopeVector;
310  delete[] secondSlopeVector;
311 
312  // and, finally, we return the cut point.
313  return max;
314  }

318  // which means the end of the current scope.

341  act++;

345 
346  // We look for a first point to first slope.
347  do {
348  act++;
349  } while ((numColumnsWithGaps[act]) == 0);
350  prev = act;
351  if ((act + 1) >= maxIter)
352  break;
353 
354  // We look for a second point to first and second slope.
355  do {
356  act++;
357  } while ((numColumnsWithGaps[act]) == 0);
358  if (act >= maxIter)
359  break;
360 
361  // Calculate the second slope between the earlier previous and current points.
362  secondSlopeVector[act] = ((float) (act - pprev) / alig->numberOfSequences);
363  secondSlopeVector[act] /= ((float) (numColumnsWithGaps[act] + numColumnsWithGaps[prev]) / alig->originalNumberOfResidues);
364 
365  // If the ratio between second slope current and second slope earlier previous points.
366  if (secondSlopeVector[pprev] != -1.0) {
367  if ((secondSlopeVector[act] / secondSlopeVector[pprev]) > maxSlope) {
368  maxSlope = (secondSlopeVector[act] / secondSlopeVector[pprev]);
369  max = pprev;
370  }
371  } else if (secondSlopeVector[prev] != -1.0) {
372  if ((secondSlopeVector[act] / secondSlopeVector[prev]) > maxSlope) {
373  maxSlope = (secondSlopeVector[act] / secondSlopeVector[prev]);
374  max = pprev;
375  }
376  }
377  act = prev;
378  }
379 
380  // We deallocate local memory.
381  delete[] secondSlopeVector;
382 
383  // Finally, we return the selected cut point.
