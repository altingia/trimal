241  int Consistency::compareAndChoose(Alignment **vectAlignments,
242  char **fileNames,
243  float *columnsValue,
244  int numAlignments,
245  bool verbosity) {
246  // Create a timerLevel that will report times upon its destruction
247  // which means the end of the current scope.
248  StartTiming("int Consistency::compareAndChoose("
249  "Alignment **vectAlignments, "
250  "char **fileNames, "
251  "float *columnsValue, "
252  "int numAlignments, "
253  "bool verbosity) ");
254 
255  int *numResiduesAlig, *correspNames, *columnSeqMatrix, *columnSeqMatrixAux;
256  int i, k, l, j, m, numSeqs, pairRes, hits, alignmentIndex = 0;
257  float max = 0, value = 0, **vectHits;
258  bool appearErrors = false;
259  string *names;
260 
261 
262  // Get some parameters from the alignment that has
263  // been selected */
264  numSeqs = vectAlignments[0]->getNumSpecies();
265 
266  // Allocate dinamic local memory
267  names = new string[numSeqs];
268  correspNames = new int[numSeqs];
269  numResiduesAlig = new int[numAlignments];
270  columnSeqMatrix = new int[numSeqs];
271  vectHits = new float *[numAlignments];
272  columnSeqMatrixAux = new int[numSeqs];
273 
274  // Check that all of alignment has the same number of
275  // sequence as well as there exists a correspondence
276  // between the names for each pars of aligs.
277  for (i = 1; i < numAlignments; i++) {
278 
279  if (numSeqs != vectAlignments[i]->getNumSpecies()) {
280  debug.report(ErrorCode::DifferentNumberOfSequencesInCompareset);
281  appearErrors = true;
282  break;
283  }
284 
285  vectAlignments[i]->getSequences(names);
286  if (!vectAlignments[0]->getSequenceNameOrder(names, correspNames)) {
287  debug.report(ErrorCode::DifferentSeqsNamesInCompareset);
288  appearErrors = true;
289  break;
290  }
291  }
292 
293  if (!appearErrors) {
294 
295  max = 0;
296 
297  // Changes the order in sequences number matrix
298  // according to the order in the selected alignment
299  for (i = 1; i < numAlignments; i++) {
300  vectAlignments[i]->getSequences(names);
301  vectAlignments[0]->getSequenceNameOrder(names, correspNames);
302  vectAlignments[i]->SequencesMatrix->setOrder(correspNames);
303  }
304 
305  // Get back the residues number for each alignment
306  for (i = 0; i < numAlignments; i++)
307  numResiduesAlig[i] = vectAlignments[i]->getNumAminos();
308 
309  // Start the comparison among the alignments
310  for (i = 0; i < numAlignments; i++) {
311  value = 0;
312  // If it's necessary, we print some information
313  if (verbosity)
314  cout << endl;
315 
316  // Initialize the hits vector for each alignment
317  vectHits[i] = new float[numResiduesAlig[i]];
318  utils::initlVect(vectHits[i], numResiduesAlig[i], 0);
319 
320  for (j = 0, pairRes = 0, hits = 0; j < numResiduesAlig[i]; j++, pairRes = 0, hits = 0) {
321 
322 
323  // Get back each column from the current selected
324  // alignment
325  vectAlignments[i]->SequencesMatrix->getColumn(j, columnSeqMatrix);
326 
327  // For each position from the previous recovered
328  // columns, we carry out the analysis to detect the
329  // same residues pair in the rest of the alignmetns
330  for (k = 0; k < numSeqs; k++) {
331 
332  // If there is a valid residue, we go ahead with
333  // the analysis
334  if (columnSeqMatrix[k] != 0) {
335  for (l = 0; l < i; l++) {
336  // Recover the residue pairs from the others aligs
337  vectAlignments[l]->SequencesMatrix->getColumn(columnSeqMatrix[k], k, columnSeqMatrixAux);
338  // and count the similar residue pairs
339  for (m = k + 1; m < numSeqs; m++)
340  if (columnSeqMatrix[m] != 0) {
341  if (columnSeqMatrix[m] == columnSeqMatrixAux[m])
342  hits++;
343  pairRes++;
344  }
345  }
346 
347  for (l = i + 1; l < numAlignments; l++) {
348  // Recover the residue pairs from the others aligs
349  vectAlignments[l]->SequencesMatrix->getColumn(columnSeqMatrix[k], k, columnSeqMatrixAux);
350  // and count the similar residue pairs
351  for (m = k + 1; m < numSeqs; m++)
352  if (columnSeqMatrix[m] != 0) {
353  if (columnSeqMatrix[m] == columnSeqMatrixAux[m])
354  hits++;
355  pairRes++;
356  }
357  }
358  }
359  }
360 
361  // For each column, compute the hits proportion for
362  // every residue pair against the rest of alignments
363  if (pairRes != 0) {
364  vectHits[i][j] += ((1.0 * hits) / pairRes);
365  value += vectHits[i][j];
366  }
367  }
368 
369  // The method can offer some information about the
370  // comparison progression
371  if (verbosity)
372  cout << "File:\t\t" << fileNames[i] << endl << "Values:\t\tSequences: " << numSeqs
373  << "\tResidues: " << numResiduesAlig[i] << "\tPond. Hits: " << setw(8)
374  << value << "\t%Consistency: " << value / numResiduesAlig[i] << endl;
375  // Keep the alignment with higher consistency value
376  if ((value / numResiduesAlig[i]) > max) {
377  alignmentIndex = i;
378  max = value / numResiduesAlig[i];
379  }
380  }
381 
382  // Prints the alignment that have been selected
383  if (verbosity) {
384  cout << "\t\t\t\t\t--------------" << endl;
385  cout << endl << "File Selected:\t" << fileNames[alignmentIndex] << endl << "Value:\t\t" << max << endl << endl;
386  }
387 
388  // The method returns a vector with the consistency
389  // value for each column in the selected alignment
390  if (columnsValue != nullptr) {
391  utils::initlVect(columnsValue, numResiduesAlig[alignmentIndex], -1);
392  for (i = 0; i < numResiduesAlig[alignmentIndex]; i++)
393  columnsValue[i] = vectHits[alignmentIndex][i];
394  }
395  }
396 
397  // Deallocate memory
398  for (i = 0; i < numAlignments; i++)
399  delete[] vectHits[i];
400 
401  delete[] vectHits;
402  delete[] names;
403  delete[] correspNames;
404  delete[] numResiduesAlig;
405  delete[] columnSeqMatrix;
406  delete[] columnSeqMatrixAux;
407 
408  // Return the selected alignment index or an error
409  // flag otherwise
410  if (appearErrors) return -1;
411  else return alignmentIndex;
412  }

416  bool Consistency::forceComparison(Alignment **vectAlignments,
417  int numAlignments,
418  Alignment *selected,
419  float *columnsValue) {
420  // Create a timerLevel that will report times upon its destruction
421  // which means the end of the current scope.
422  StartTiming("bool Consistency::forceComparison("
423  "Alignment **vectAlignments, "
424  "int numAlignments, "
425  "Alignment *selected, "
426  "float *columnsValue) ");
427 
428  int *correspNames, *columnSeqMatrix, *columnSeqMatrixAux;
429  int i, j, k, ll, numResidues, numSeqs, pairRes, hit;
430  bool appearErrors = false;
431  string *names;
432 
433  // Get some parameters from the alignment that has
434  // been selected
435  numResidues = selected->getNumAminos();
436  numSeqs = selected->getNumSpecies();
437 
438  // Initialize the vector where we are going to store
439  // the proportion of hits for each column in the
440  // selected alignment
441  utils::initlVect(columnsValue, numResidues, 0);
442 
443  // Allocate dynamic local memory
444  names = new string[numSeqs];
445  correspNames = new int[numSeqs];
446  columnSeqMatrix = new int[numSeqs];
447  columnSeqMatrixAux = new int[numSeqs];
448  // Check that all of alignment has the same number of
449  // sequence as well as there exists a correspondence
450  // between the names for each pars of aligs.
451  for (i = 0; i < numAlignments; i++) {
452 
453  if (numSeqs != vectAlignments[i]->getNumSpecies()) {
454  debug.report(ErrorCode::DifferentNumberOfSequencesInCompareset);
455  appearErrors = true;
456  break;
457  }
458 
459  vectAlignments[i]->getSequences(names);
460  if (!selected->getSequenceNameOrder(names, correspNames)) {
461  debug.report(ErrorCode::DifferentSeqsNamesInCompareset);
462  appearErrors = true;
463  break;
464  }
465  }
466  // Changes the order in sequences number matrix
467  // according to the order in the selected alignment
468  for (i = 0; i < numAlignments; i++) {
469  vectAlignments[i]->getSequences(names);
470  selected->getSequenceNameOrder(names, correspNames);
471  vectAlignments[i]->SequencesMatrix->setOrder(correspNames);
472  }
473 
474  if (selected->SequencesMatrix == nullptr)
475  selected->SequencesMatrix = new Alignment::sequencesMatrix(selected);
476  // Do the same analysis for each column
477  for (i = 0, pairRes = 0, hit = 0;
478  i < numResidues && !appearErrors;
479  i++, pairRes = 0, hit = 0) {
480 
481  // We get back the sequence position for each residue
482  // from every column in the selected alignment
483  utils::initlVect(columnSeqMatrix, numSeqs, 0);
484  selected->SequencesMatrix->getColumn(i, columnSeqMatrix);
485  // For each residue pairs, we look for it in the rest
486  // of alignments
487  for (j = 0; j < numSeqs; j++) {
488 
489  // If there is a residue, not a gap, we carry out the
490  // assesment
491  if (columnSeqMatrix[j] != 0) {
492  for (k = 0; k < numAlignments; k++) {
493 
494  // We look for the same residue in the same row in
495  // the rest of alignments
496  utils::initlVect(columnSeqMatrixAux, numSeqs, 0);
497  vectAlignments[k]->SequencesMatrix->getColumn(columnSeqMatrix[j], j, columnSeqMatrixAux);
498  // We count when we get the same residue pairs in the
499  // rest of alignments
500  for (ll = j + 1; ll < numSeqs; ll++)
501  if (columnSeqMatrix[ll] != 0) {
502  if (columnSeqMatrix[ll] == columnSeqMatrixAux[ll])
503  hit++;
504  pairRes++;
505  }
506  }
507  }
508  }
509  // Store the hits proportion for each column
510  if (pairRes != 0) columnsValue[i] += ((1.0 * hit) / pairRes);
511  }
512 
513  // Deallocate dynamic memory
514  delete[] names;
515  delete[] correspNames;
516  delete[] columnSeqMatrix;
517  delete[] columnSeqMatrixAux;
518 
519  // Return if the process was perform without errors
520  return !appearErrors;
521 
522  }

524 // This method applies a specific windows size to a selected alignment
525  bool Consistency::applyWindow(int halfW) {
526  // Create a timerLevel that will report times upon its destruction
527  // which means the end of the current scope.
528  StartTiming("bool Consistency::applyWindow(int columns, int halfWindowApplied, float *columnsValue) ");
529 
530  if (halfW > residues / 4) {
531  debug.report(ErrorCode::ConsistencyWindowTooBig);
532  return false;
533  }
534 
535  // If the current half window is the same as the last one, don't do anything
536  if (halfWindow == halfW) return true;
537 
538  // Save the requested half window. This is useful when making a copy of the
539  // alignment, as the window values are not valid anymore but don't want to
540  // calculate them if not needed anymore
541  halfWindow = halfW;
542 
543  // If the half window requested is 0 or a negative number
544  // we simply delete the window values.
545  if (halfW < 1) {
546  delete[] values_windowed;
547  values_windowed = nullptr;
548  return true;
549  }
550 
551  // Initialize the values used in the calculation
552  int i, j, window;
553 
554  // Initialize the consistency window array if it's null
555  if (values_windowed == nullptr)
556  values_windowed = new float[residues];
557 
558 
559  window = 2 * halfWindow + 1;
560 
561  // Do the average window calculations
562  for (i = 0; i < residues; i++) {
563  values_windowed[i] = 0.F;
564  for (j = i - halfWindow; j <= i + halfWindow; j++) {
565  if (j < 0)
566  values_windowed[i] += values[-j];
567  else if (j >= residues)
568  values_windowed[i] += values[((2 * residues - j) - 2)];
569  else
570  values_windowed[i] += values[j];
571  }
572 
573  // Calculate the similarity value for the i column
574  values_windowed[i] = values_windowed[i] / window;
575  }
576 
577  return true;
578  }
