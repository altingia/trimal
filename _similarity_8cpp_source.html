97  void Similarity::calculateMatrixIdentity() {
98  // Create a timerLevel that will report times upon its destruction
99  // which means the end of the current scope.
100  StartTiming("void Similarity::calculateMatrixIdentity() ");
101 
102  // We don't want to calculate the matrix identity
103  // if it has been previously calculated
104  if (matrixIdentity != nullptr)
105  return;
106 
107  // Allocate temporal variables
108  char indet;
109  int i, j, k;
110  int sum, length;
111 
112  // Allocate memory for the matrix identity
113  matrixIdentity = new float *[alig->originalNumberOfSequences];
114  for (i = 0; i < alig->originalNumberOfSequences; i++) {
115  matrixIdentity[i] = new float[alig->originalNumberOfSequences];
116  }
117 
118  // Depending on alignment type, indetermination symbol will be one or other
119  indet = alig->getAlignmentType() & SequenceTypes::AA ? 'X' : 'N';
120 
121  // For each sequences' pair
122  for (i = 0; i < alig->originalNumberOfSequences; i++) {
123  for (j = i + 1; j < alig->originalNumberOfSequences; j++) {
124  // For each position in the alignment of that pair than we are processing
125  for (k = 0, sum = 0, length = 0; k < alig->originalNumberOfResidues; k++) {
126  // If we find a element that is not a gap or an X aminoacid in the first sequence of the pair
127  if ((alig->sequences[i][k] != '-') &&
128  (alig->sequences[i][k] != indet)) {
129 
130  // If we also find a valid element in the second sequence
131  if ((alig->sequences[j][k] != '-') &&
132  (alig->sequences[j][k] != indet))
133  {
134  // If the two valid elements are the same increase the sum
135  if (alig->sequences[j][k] == alig->sequences[i][k])
136  sum++;
137  }
138  // Increase the length of the sequence free of gaps and X elements
139  length++;
140  }
141  // If the first processed element is invalid and in the second we find a valid element increase the length of
142  // the sequence free of gaps and X elements
143  else if ((alig->sequences[j][k] != '-') &&
144  (alig->sequences[j][k] != indet))
145  {
146  length++;
147  }
148  }
149 
150  // Calculate the value of matrix idn for columns j and i
151  matrixIdentity[j][i] = (1.0F - ((float)sum / length) );
152  matrixIdentity[i][j] = matrixIdentity[j][i];
153  }
154  }
155  }

157  bool Similarity::calculateVectors(bool cutByGap) {
158  // Create a timerLevel that will report times upon its destruction
159  // which means the end of the current scope.
160  StartTiming("bool Similarity::calculateVectors(int *gaps) ");
161 
162  // A similarity matrix must be defined. If not, return false
163  if (simMatrix == nullptr)
164  return false;
165 
166  // Calculate the matrix identity in case it's not done before
167  if (matrixIdentity == nullptr)
168  calculateMatrixIdentity();
169 
170  // Create the variable gaps, in case we want to cut by gaps
171  int *gaps = nullptr;
172 
173  // Retrieve the gaps values in case we want to set to 0 the similarity value
174  // in case the gaps value for that column is bigger or equal to 0.8F
175  if (cutByGap)
176  {
177  if (alig->Statistics->gaps == nullptr)
178  alig->Statistics->calculateGapStats();
179  gaps = alig->Statistics->gaps->getGapsWindow();
180  }
181 
182  // Initialize the variables used
183  int i, j, k;
184  float num, den;
185 
186  // Depending on alignment type, indetermination symbol will be one or other
187  char indet = alig->getAlignmentType() & SequenceTypes::AA ? 'X' : 'N';
188 
189  // Q temporal value
190  float Q;
191  // Temporal chars that will contain the residues to compare by pair.
192  char chA, chB;
193 
194  // Calculate the maximum number of gaps a column can have to calculate it's
195  // similarity
196  float gapThreshold = 0.8F * alig->numberOfResidues;
197 
198  // For each column calculate the Q value and the MD value using an equation
199  for (i = 0; i < alig->originalNumberOfResidues; i++) {
200  // Set MDK for columns with gaps values bigger or equal to 0.8F
201  if (cutByGap && gaps[i] >= gapThreshold) {
202  MDK[i] = 0.F;
203  continue;
204  }
205  // For each AAs/Nucleotides' pair in the column we compute its distance
206  for (j = 0, num = 0, den = 0; j < alig->originalNumberOfSequences; j++) {
207 
208 
209  // Calculate the upper value of the residue,
210  // to use in simMatrix->getDistance
211  // This is faster than calculating the upper on that method
212  // as this is done before entering the loop
213  // Doing this before checking if the element is indeterminate or gap
214  // allows to check if the indetermination is not capitalized
215  chA = utils::toUpper(alig->sequences[j][i]);
216 
217  // We don't compute the distance if the first element is
218  // a indeterminate (XN) or a gap (-) element.
219  if ((chA == '-') || (chA == indet))
220  continue;
221 
222  for (k = j + 1; k < alig->originalNumberOfSequences; k++) {
223  // We calculate the upper value of the residue,
224  // to use in simMatrix->getDistance
225  // This is equally faster as if it was done inside the method
226  // but to prevent errors, the method doesn't 'upper'
227  // the given chars.
228  // Doing this before checking if the element is indeterminate or gap
229  // allows to check if the indetermination is not capitalized
230  chB = utils::toUpper(alig->sequences[k][i]);
231 
232  // We don't compute the distance if the second element is
233  // a indeterminate (XN) or a gap (-) element
234  if ((chB == '-') || (chB == indet))
235  continue;
236 
237  // We use the identity value for the two pairs and
238  // its distance based on similarity matrix's value.
239  num += matrixIdentity[j][k] * simMatrix->getDistance(chA, chB);
240  den += matrixIdentity[j][k];
241  }
242  }
243 
244  // If we are processing a column with only one AA/nucleotide, MDK = 0
245  if (den == 0)
246  MDK[i] = 0;
247  else
248  {
249  Q = num / den;
250  // If the MDK value is more than 1, we normalized this value to 1.
251  // Only numbers higher than 0 yield exponents higher than 1
252  // Using this we can test if the result is going to be higher than 1.
253  // And thus, prevent calculating the exp.
254  // Take in mind that the Q is negative, so we must test if Q is LESSER
255  // than one, not bigger.
256  if (Q < 0)
257  MDK[i] = 1.F;
258  else
259  MDK[i] = exp(-Q);
260  }
261  }
262 
263  for (i = 0; i < alig->originalNumberOfSequences; i++)
264  delete[] matrixIdentity[i];
265  delete[] matrixIdentity;
266  matrixIdentity = nullptr;
267 
268  return true;
269  }

271  bool Similarity::applyWindow(int halfW) {
272  // Create a timerLevel that will report times upon its destruction
273  // which means the end of the current scope.
274  StartTiming("bool Similarity::applyWindow(int halfW) ");
275 
276  // Calculate the MDK array if it has not been calculated previously
277  if (MDK == nullptr)
278  calculateVectors(true);
279 
280  // Check is the half window value passed is in the valid range
281  if (halfW > alig->originalNumberOfResidues / 4) {
282  debug.report(ErrorCode::SimilarityWindowTooBig);
283  return false;
284  }
285 
286  // If the current half window is the same as the last one, don't do anything
287  if (halfWindow == halfW) return true;
288 
289  // Save the requested half window. This is useful when making a copy of the
290  // alignment, as the window values are not valid anymore but don't want to
291  // calculate them if not needed anymore
292  halfWindow = halfW;
293 
294  // If the half window requested is 0 or a negative number
295  // we simply delete the window values.
296  if (halfW < 1) {
297  delete[] MDK_Window;
298 
299  MDK_Window = nullptr;
300  return true;
301  }
302 
303  // Initialize the values used in the calculation
304  int i, j, window;
305 
306  // Initialize the MDK window array if it's null
307  if (MDK_Window == nullptr)
308  MDK_Window = new float[alig->originalNumberOfResidues + 1];
309 
310  window = 2 * halfWindow + 1;
311 
312  // Do the average window calculations
313  for (i = 0; i < alig->originalNumberOfResidues; i++) {
314  MDK_Window[i] = 0.F;
315  for (j = i - halfWindow; j <= i + halfWindow; j++) {
316  if (j < 0)
317  MDK_Window[i] += MDK[-j];
318  else if (j >= alig->originalNumberOfResidues)
319  MDK_Window[i] += MDK[((2 * alig->originalNumberOfResidues - j) - 2)];
320  else
321  MDK_Window[i] += MDK[j];
322  }
323 
324  // Calculate the average value, by dividing the values
325  MDK_Window[i] = MDK_Window[i] / (float) window;
326  }
327  return true;
328  }

381  double Similarity::calcCutPoint(float baseLine, float conservationPct) {
382  // Create a timerLevel that will report times upon its destruction
383  // which means the end of the current scope.
384  StartTiming("double Similarity::calcCutPoint(float baseLine, float conservationPct) ");
385  // It computes the cutting point based on alignment's similarity values -
386  // the so-called 'similarity'. It also takes into account the minimum percentage
387  // from the input alignment to be kept. Depending on those two values, the
388  // method will select a different cutting-point.
389 
390  double cuttingPoint_MinimumConserv, cuttingPoint_SimilThreshold;
391  int i, highestPos;
392  float *vectAux;
393 
394  vectAux = new float[alig->originalNumberOfResidues];
395 
396  // Sort a copy of the vector containing the similarity values after applying
397  // any windows methods. Take the columns value that it lower than the minimum
398  // similarity threshold set by the user
399  utils::copyVect(getMdkWindowedVector(), vectAux, alig->originalNumberOfResidues);
400  utils::quicksort(vectAux, 0, alig->originalNumberOfResidues - 1);
401 
402  for (i = alig->originalNumberOfResidues - 1; i >= 0; i--)
403  if (vectAux[i] < conservationPct)
404  break;
405  cuttingPoint_SimilThreshold = vectAux[i];
406 
407  // It is possible that due to number casting, we get a number out of the
408  // vector containing the similarity values - it is not reporting an overflow
409  // situation but giving back a 0 when it should be a number equal (or closer)
410  // to 1.
411  highestPos = (int) ((double) (alig->originalNumberOfResidues - 1) * (100.0 - baseLine) / 100.0);
412  highestPos = highestPos < (alig->originalNumberOfResidues - 1) ? highestPos : alig->originalNumberOfResidues - 1;
413  cuttingPoint_MinimumConserv = vectAux[highestPos];
414 
415  delete[] vectAux;
416 
417  // Return the minimum cutting point between the one set by the threshold and
418  // the one set by the minimum percentage of the input alignment to be kept
419  return (cuttingPoint_MinimumConserv < cuttingPoint_SimilThreshold ?
420  cuttingPoint_MinimumConserv : cuttingPoint_SimilThreshold);
421  }
