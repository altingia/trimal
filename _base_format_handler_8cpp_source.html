68  std::string FormatManager::replaceINtag(
69  const Alignment & alignment,
70  const std::string & outPattern)
71  {
72  // Replace the [in] tag with correspondent information.
73  unsigned long start, end;
74  // Although an alignment file should have a filename
75  // we make a guard to prevent issues.
76  if (alignment.filename.empty())
77  {
78  return utils::ReplaceString(outPattern, "[in]", "NoInputFileName");
79  }
80  // If the alignment contains a filename

85  end = alignment.filename.find_last_of('.');
86  return utils::ReplaceString(outPattern, "[in]", alignment.filename.substr(start, end-start));

99  bool FormatManager::saveAlignments(
100  const std::string &outPattern,
101  const std::vector<std::string> &outFormats,
102  const std::vector<Alignment *> &alignmentVector)
103  {
104  StartTiming("bool FormatManager::saveAlignment()");
105 
106  bool returnValue = true;
107 
108  std::vector<FormatHandling::BaseFormatHandler*> formats{};
109 
110  for(const std::string & outFormat : outFormats)
111  {
112  // Get the format desired
113  BaseFormatHandler* state = getFormatFromToken(outFormat);
114  if (state == nullptr)
115  {
116  debug.report(ErrorCode::OutputFormatNotRecognized, &outFormats.at(0)[0]);
117  returnValue = false;
118  continue;
119  }
120  formats.emplace_back(state);
121  }
122 
123 
124  // Check if any format has been requested.
125  // If not, the default is fasta.
126  if (formats.empty())
127  formats.emplace_back(getFormatFromToken("fasta"));
128 
129 
130  // If we are outputting to terminal
131  if (outPattern.empty())
132  {
133  if (formats.size() != 1)
134  {
135  // Report an error if more than one format is requested on terminal
136  debug.report(ErrorCode::OnlyOneFormatOnConsoleOutput);
137  return false;
138  }
139 
140  return formats[0]->SaveAlignment(*alignmentVector[0], &std::cout);
141  }
142 
143  // Iterate over all alignments
144  for (const Alignment * alignment : alignmentVector)
145  {
146  // Check if is possible to save the alignment
147  if (alignment->numberOfResidues == 0 ||
148  alignment->numberOfSequences == 0)
149  {
150  debug.report(ErrorCode::AlignmentIsEmpty);
151  returnValue = false;
152  continue;
153  }
154 
155  // Retrieve the IN parameter
156  std::string filename = replaceINtag(*alignment, outPattern);
157 
158  for(FormatHandling::BaseFormatHandler * state : formats)
159  {
160  // Replace the [extension] and [format] tokens
161  std::string finalFilename =
162  utils::ReplaceString(filename, "[extension]", state->extension);
163  utils::ReplaceStringInPlace(finalFilename, "[format]", state->name);
164 // Macro passed as compile time. See FormatHandlerOverwritePolicy.cmake
165 #if FormatHandlerOverwrites && false
166  if (utils::fileExists(finalFilename) && (openmode & std::ofstream::app) == 0)
167  {
168  debug.report(OverwrittingFile, new std::string[2]{state->name, finalFilename} );
169  }
170 #else
171  if((openmode & std::ofstream::app) == 0)
172  {
173 
174  uint i;
175  for (i = 0; i < std::numeric_limits<uint>::max(); i++)
176  {
177  if (!utils::fileExists(finalFilename + "." + std::to_string(i)))
178  {
179  debug.report(RenamingOutputPreventOverride,
180  new std::string[3]{state->name, finalFilename, finalFilename + "." + std::to_string(i)} );
181  finalFilename += "." + std::to_string(i);
182  break;
183  }
184  }
185  if (i == std::numeric_limits<uint>::max())
186  {
187 // Macro passed as compile time. See FormatHandlerOverwritePolicy.cmake
188 #if FormatHandlerOverwritesOriginal
189  debug.report(TriedRenamingOutputPreventOverride,
190  new std::string[3]{state->name, finalFilename, finalFilename} );
191 #else
192  debug.report(TriedRenamingOutputPreventOverride,
193  new std::string[3]{state->name, finalFilename, finalFilename + "." + std::to_string(i)} );
194  finalFilename += "." + std::to_string(i);
195 #endif
196  }
197 #endif
198  }
199 
200  // Open the file handler
201  std::ofstream outFileHandler(finalFilename, openmode);
202 
203  // Save the alignment
204  if (!state->SaveAlignment(*alignment, &outFileHandler))
205  {
206  debug.report(ErrorCode::AlignmentNotSaved, &state->name[0]);
207  returnValue = false;
208  }
209  }
210  }
211 
212  return returnValue;
213  }

302  FormatHandling::BaseFormatHandler * FormatManager::getFormatFromFile(
303  const std::string &filename)
304  {
305  // Open the file
306  std::ifstream * input = getNonEmptyFile(filename);
307 
308  if (input == nullptr)
309  {
310  delete input;
311  return nullptr;
312  }
313 
314  // Temporal value
315  BaseFormatHandler* inState = nullptr;
316 
317  // Compare variables
318  int format_value = 0, temp_value = 0;
319 
320  // Iterate over all available states
321  for(BaseFormatHandler* state : available_states)
322  {
323  // Store the score that state produces
324  temp_value = state -> CheckAlignment(input);
325 
326  // If the score is better than last maximum, this format
327  // is better at recognizing the input file
328  if (temp_value > format_value)
329  {
330  format_value = temp_value;
331  inState = state;
332  }
333  }
334 
335  // We don't need the input open anymore. Destructor deletes it
336  delete input;
337 
338  // Check if any input recognized the format
339  if (inState == nullptr)
340  {
341  debug.report(ErrorCode::AlignmentFormatNotRecognized, &filename[0]);
342  return nullptr;
343  }
344 
345  // Return the recognized format
346  return inState;
347  }
