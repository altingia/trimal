60  bool Mold::applyWindow(int halfW) {
61  // Check is the half window value passed is in the valid range
62  if (halfW > residues / 4) {
63  debug.report(ErrorCode::WindowTooBig);
64  return false;
65  }
66 
67  // Calculate the values array if it has not been calculated previously
68  if (values == nullptr) {
69  calculate();
70  }
71 
72  // If the current half window is the same as the last one, don't do anything
73  if (halfWindowApplied == halfW) return true;
74 
75  // If the half window requested is 0 or a negative number
76  // we simply delete the window values.
77  if (halfW < 1) {
78  if (halfWindowApplied > 0)
79  delete[] valuesWindow;
80 
81  valuesWindow = nullptr;
82  return true;
83  }
84 
85  // Initialize the values used in the calculation
86  int i, j;
87 
88  // Initialize the values window array if it's null
89  if (valuesWindow == nullptr)
90  valuesWindow = new float[residues];
91 
92  // Special value. We don't compute the average
93  // if there is no neighbour on one side
94  valuesWindow[0] = values[0];
95  int currentWindowSize = 1;
96 
97  // First loop: While index is lower than the window value,
98  // We apply a window as big as the index
99  for (i = 1; i < halfW + 1; i++) {
100 
101  // Increase the window size by 2, one residue on each side.
102  currentWindowSize += 2;
103 
104  valuesWindow[i] = 0.F;
105 
106  for (j = 0; j < currentWindowSize; j++) {
107  valuesWindow[i] += values[j];
108  }
109 
110  valuesWindow[i] = valuesWindow[i] / currentWindowSize;
111  }
112 
113  // Second loop: Apply requested window size until
114  // i + windowsize > array size
115  {
116  float window = 2 * halfW + 1;
117 
118  for (; i != residues - halfW; i++) {
119 
120  valuesWindow[i] = values[i];
121 
122  for (j = 1; j < halfW + 1; j++) {
123  valuesWindow[i] += values[i + j];
124  valuesWindow[i] += values[i - j];
125  }
126 
127  valuesWindow[i] = valuesWindow[i] / window;
128  }
129  }
130 
131  // Third loop: As index is bigger than the number of residues - halfwindow
132  // Apply a symetric window as big as possible
133  for (; i < residues - 1; i++) {
134 
135  // Decrease the window size by 2, one residue on each side.
136  currentWindowSize -= 2;
137 
138  valuesWindow[i] = values[i];
139 
140  for (j = 1; (i + j) < residues; j++) {
141  valuesWindow[i] += values[i + j];
142  valuesWindow[i] += values[i - j];
143  }
144 
145  // Calculate the average value, by dividing the current value (sumatory)
146  // by the current window size
147  valuesWindow[i] = valuesWindow[i] / currentWindowSize;
148 
149  }
150 
151  // Special value. We don't compute the average
152  // if there is no neighbour on one side
153  valuesWindow[residues - 1] = values[residues - 1];
154 
155  // New window has been applied
156  halfWindowApplied = halfW;
157 
158  return true;
159  }
